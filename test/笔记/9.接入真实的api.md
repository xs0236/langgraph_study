import requests

def get_wechat_token():
appid = os.getenv("WECHAT_APPID")
secret = os.getenv("WECHAT_SECRET")
url = f"https://api.weixin.qq.com/cgi-bin/token?grant_type=client_credential&appid={appid}&secret={secret}"
return requests.get(url).json().get("access_token")

def p_wechat(state: PublishState): # ... 生成文案部分不变 ...

    # ==================== 改成真实 API ====================
    access_token = get_wechat_token()

    # 1. 上传图片素材（如果有图）
    # 2. 创建草稿
    draft_url = f"https://api.weixin.qq.com/cgi-bin/draft/add?access_token={access_token}"
    data = {
        "articles": [{
            "title": state['topic'],
            "content": state['article'].replace("\n", "<br>"),
            "digest": wechat_content[:120],
            "author": "AI团队"
        }]
    }
    resp = requests.post(draft_url, json=data).json()

    if "media_id" in resp:
        wechat_url = f"草稿创建成功，Media ID: {resp['media_id']}"
    else:
        wechat_url = f"失败: {resp.get('errmsg')}"

    return {"wechat_url": wechat_url}
