import requests

def ragflow_retrieve(query, dataset_id, top_k=5):
"""从 RAGFlow 知识库检索相关片段"""
url = "http://localhost:9380/api/retrieval"
headers = {"Authorization": "ragflow-your-api-key"}
payload = {
"dataset_id": dataset_id,
"question": query,
"top_k": top_k,
"similarity_threshold": 0.5,
"rerank": True # 启用重排序
}
resp = requests.post(url, json=payload, headers=headers)
chunks = resp.json()["data"]["chunks"]
return format_chunks(chunks)
