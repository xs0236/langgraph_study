错误定位
plain
langgraph.errors.InvalidUpdateError: Expected dict, got [Send(...), Send(...)]
这个错误发生在 翻译派发节点 trans_fan_out，当用户输入 continue 后，主调度器将 next_agent 路由到 trans_fan_out，但 trans_fan_out 返回的是 list[Send]，而 LangGraph 期望它返回一个 dict 来更新状态。
根本原因：trans_fan_out 的返回类型与 LangGraph 状态更新机制冲突
在 LangGraph 中，普通节点（非 Send 的目标节点）必须返回 dict 来更新状态。但 trans_fan_out 使用 Send 进行动态派发（fan-out），它的返回值是 list[Send]，这是 conditional edge 的 mapper 函数 的返回格式，不是普通节点的返回格式。
你的代码结构有问题：
Python

trans_fan_out 被同时当作了节点和 edge mapper，导致冲突。

正确的 LangGraph Send / Fan-out 模式
在 LangGraph 中，Send 只能在 conditional edge 中使用，不能作为节点的返回值。
正确的模式是：

# 2. do_translate 是一个普通节点，返回 dict（可以设置一个标志）

def do_translate(state): # 这个节点其实什么都不做，只是作为中转 # 真正的 fan-out 通过 conditional edge 从这个节点出发

# 3. 从 do_translate 节点出发的 conditional edge 使用 Send

def route_translate(state):

sends.append(Send("translate_one", {"lang": "en", ...}))
if not state.get("japanese_version"):
sends.append(Send("translate_one", {"lang": "ja", ...}))

mg.add_node("do_translate", do_translate)
mg.add_conditional_edges("do_translate", route_translate, ["translate_one"])

def m_s(s):
features = s.get("features", {})

    if not s.get("article"):
        return {"next_agent": "writing"}

    if s.get("article") and s.get("ai_quality_score") == 0:
        return {"next_agent": "ai_quality_check"}

    if s.get("ai_quality_score", 10) < 7 and s.get("human_decision") != "continue":
        print("  🔄 [主调度] AI自检评分 {} 低于7分，自动重写".format(s["ai_quality_score"]))
        return {
            "next_agent": "writing",
            "human_decision": "",
            "article": "",
            "edit_feedback": "",
            "ai_quality_score": 0,
            "ai_quality_issues": "",
            "previous_feedback": "AI自检发现问题，请改进：{}".format(s.get("ai_quality_issues", ""))
        }

    if s.get("article") and not s.get("human_decision"):
        return {"next_agent": "human_review"}

    decision = s.get("human_decision", "")

    if decision == "stop":
        return {"next_agent": "FINISH"}

    if decision != "continue" and decision:
        print("  🔄 [主调度] 用户要求重写，意见：{}".format(decision))
        return {
            "next_agent": "writing",
            "human_decision": "",
            "article": "",
            "edit_feedback": "",
            "ai_quality_score": 0,
            "ai_quality_issues": "",
            "previous_feedback": decision,
            "translations": [],
            "english_version": "",
            "japanese_version": "",
            "image_prompt": "",
            "image_url": "",
            "wechat_url": "",
            "zhihu_url": "",
            "weibo_text": "",
            "ai_need_image": None,
            "ai_image_reason": "",
            "ai_publish_platforms": "",
            "ai_publish_styles": ""
        }

    # ========== 修改这里：翻译逻辑 ==========
    if features.get("translate"):
        if not s.get("english_version") or not s.get("japanese_version"):
            return {"next_agent": "translate"}  # 改为路由到 translate 节点

    if features.get("image"):
        if s.get("ai_need_image") is None:
            return {"next_agent": "ai_image_decision"}
        if s.get("ai_need_image") and not s.get("image_url"):
            return {"next_agent": "image"}

    if features.get("publish"):
        if not s.get("ai_publish_platforms"):
            return {"next_agent": "ai_publish_plan"}
        platforms_str = s.get("ai_publish_platforms", "")
        platforms = [p.strip() for p in platforms_str.split(",") if p.strip()]
        need_pub = False
        if "wechat" in platforms and not s.get("wechat_url"): need_pub = True
        if "zhihu" in platforms and not s.get("zhihu_url"): need_pub = True
        if "weibo" in platforms and not s.get("weibo_text"): need_pub = True
        if need_pub:
            return {"next_agent": "publish"}

    return {"next_agent": "FINISH"}
